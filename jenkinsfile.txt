pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/Avci-10/projet_java.git'
            }
        }

        stage('Build Docker Image') {
            steps {
                sh 'docker build -t hkiriiheb/v1  .'
            }
        }

        stage('Push Docker Image') {
            steps {
                    sh 'docker login -u hkiriheb -p github123@@'
                    sh 'docker push  hkiriiheb/v1'
                
            }
        }

      
    }
}
